plot_IL1B <-
  ggplot(
    data_IL1B,
    aes(
      x = conc,
      y = time_means
    )
  ) +
  ## First layer ##
  geom_point(
    aes(
      fill = factor(time)

    stroke = 1,
    alpha = .7
  ) +
  scale_fill_brewer(palette = "Set2") +
